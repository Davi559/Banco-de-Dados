Comandos SQL

Criar um Banco de Dados

CREATE DATABASE nome_banco;

Criar uma Tabela

CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(100),
    data_cadastro DATE
);

Inserir Dados

INSERT INTO usuarios (nome, email, data_cadastro) 
VALUES ('João Silva', 'joao@example.com', '2025-05-09');

Consultar Dados

SELECT * FROM usuarios;

Atualizar Dados

UPDATE usuarios 
SET email = 'novo.email@example.com' 
WHERE id = 1;

Excluir Dados

DELETE FROM usuarios WHERE id = 1;

Selecionar com Condições

SELECT * FROM usuarios 
WHERE data_cadastro > '2025-01-01' 
ORDER BY nome ASC;

Agrupar Dados

SELECT COUNT(*) AS total_usuarios 
FROM usuarios;
